    int segment_from_mixed_selection(int s, int e) const {
        s = std::max(0, std::min(s, 7));
        e = std::max(s + 1, std::min(e, 8));
        int count[3] = {0, 0, 0};
        for (int i = s; i < e; ++i)
            ++count[segment_from_char_index(i)];
        int maxc = std::max({count[0], count[1], count[2]});
        int nmax = 0;
        for (int k = 0; k < 3; ++k)
            if (count[k] == maxc)
                ++nmax;
        if (nmax != 1)
            return segment_from_char_index(s);
        for (int k = 0; k < 3; ++k)
            if (count[k] == maxc)
                return k;
        return 0;
    }
    void sync_active_segment_from_entry_state() {
        int s = 0, e = 0;
        if (get_selection_bounds(s, e)) {
            if (s > e)
                std::swap(s, e);
            if (e > s) {
                active_segment_ = segment_from_mixed_selection(s, e);
                return;
            }
        }
        int pos = get_position();
        if (pos < 0)
            pos = 0;
        if (pos > 7)
            pos = 7;
        active_segment_ = segment_from_char_index(pos);
    }
    bool map_param_to_widget_xy(GdkEventButton *e, int &ix, int &iy) {

        if (!e->window || !win_widget)

             cur = gdk_window_get_parent(cur)) {
            gdk_window_coords_to_parent(cur, px, py, &px, &py);
        }
        ix = static_cast<int>(px);
        iy = static_cast<int>(py);
        return true;
    }

    int segment_from_widget_xy(double wx, double wy) {
        int xoff = 0, yoff = 0;

    void adjust_active_segment(int delta) {

            active_segment_ == 0 ? 3600 : (active_segment_ == 1 ? 60 : 1);
        const int cur_sec = static_cast<int>(get_adjustment()->get_value());
        long long ns = static_cast<long long>(cur_sec) +

        if (get_wrap()) {

                ns += 86400;

    void apply_digit_to_segment(int digit) {
        sync_active_segment_from_entry_state();
        int sec = static_cast<int>(get_adjustment()->get_value());
        int h = sec / 3600;
        int m = (sec % 3600) / 60;
        int s = sec % 60;
        const int seg = active_segment_;
        int *const p = &partial_[seg];
        auto roll = [digit](int v, int mx) {
            v = (v % 10) * 10 + digit;
            return std::min(mx, v);
        };
        if (*p == 0) {
            if (seg == 0)
                h = roll(h, 23);
            else if (seg == 1)
                m = roll(m, 59);
            else
                s = roll(s, 59);
            *p = -1;
        } else {
            if (seg == 0) {
                if (digit > 2) {
                    h = std::min(23, digit);
                    *p = -1;
                } else {
                    snapshot_sec_ = sec;
                    h = digit;
                    *p = 0;
                }
            } else if (seg == 1) {
                if (digit > 5) {
                    m = std::min(59, digit);
                    *p = -1;
                } else {
                    snapshot_sec_ = sec;
                    m = digit;
                    *p = 0;
                }
            } else if (seg == 2) {
                if (digit > 5) {
                    s = std::min(59, digit);
                    *p = -1;
                } else {
                    snapshot_sec_ = sec;
                    s = digit;
                    *p = 0;
                }
            } else {
                return;
            }
        }
        sec = h * 3600 + m * 60 + s;
        suppress_partial_clear_ = true;
        get_adjustment()->set_value(static_cast<double>(sec));
        update();
        suppress_partial_clear_ = false;
        select_region_for_segment(active_segment_);
    }

class SortableTreeView : public Gtk::TreeView {
    public:
        SortableTreeView() {
            model_property_conn_ =
                property_model().signal_changed().connect(
                    [this]() { on_tree_model_changed(); });
        }

        ~SortableTreeView() override {
            model_property_conn_.disconnect();
            for (auto &reg : sort_columns_)
                reg.clicked_conn.disconnect();
            sort_columns_.clear();
        }

        void register_sortable_column(
            Gtk::TreeViewColumn &column, int sort_model_column_id,
            const Glib::ustring &column_title = {}) {
            const int idx = column_view_index(*this, column);
            if (idx < 0)
                return;

            auto it = std::find_if(
                sort_columns_.begin(), sort_columns_.end(),
                [idx](const SortColumnReg &r) {
                    return r.tree_column_index == idx;
                });
            if (it != sort_columns_.end()) {
                it->clicked_conn.disconnect();
                sort_columns_.erase(it);
            }

            SortColumnReg reg;
            reg.tree_column_index = idx;
            reg.model_column = sort_model_column_id;
            if (!column_title.empty())
                column.set_title(column_title);
            column.set_clickable(true);
            column.set_sort_column(-1);
            column.set_sort_indicator(false);

            reg.clicked_conn = column.signal_clicked().connect(sigc::bind(
                sigc::mem_fun(*this, &SortableTreeView::on_column_clicked),
                &column, sort_model_column_id));
            sort_columns_.push_back(std::move(reg));
        }

        void clear_sort_state() {
            if (auto sortable = resolve_sortable(get_model())) {
                sortable->set_sort_column(
                    Gtk::TreeSortable::DEFAULT_UNSORTED_COLUMN_ID,
                    Gtk::SORT_ASCENDING);
            }
            active_sort_model_column_.reset();
            active_phase_.reset();
            refresh_sort_indicators();
        }

    protected:
        int column_view_index(
            const Gtk::TreeView &view, const Gtk::TreeViewColumn &column) {
            const int n = view.get_n_columns();
            for (int i = 0; i < n; ++i) {
                if (view.get_column(i) == &column)
                    return i;
            }
            return -1;
        }

        Glib::RefPtr<Gtk::TreeSortable> resolve_sortable(
            const Glib::RefPtr<Gtk::TreeModel> &model) {
            if (!model)
                return {};
            if (auto direct =
                    Glib::RefPtr<Gtk::TreeSortable>::cast_dynamic(model))
                return direct;
            if (auto filter =
                    Glib::RefPtr<Gtk::TreeModelFilter>::cast_dynamic(model)) {
                return Glib::RefPtr<Gtk::TreeSortable>::cast_dynamic(
                    filter->get_model());
            }
            return {};
        }

        void on_tree_model_changed() { clear_sort_state(); }

        void on_column_clicked(Gtk::TreeViewColumn *column,
                                                        int model_column) {
            (void)column;
            auto sortable = resolve_sortable(get_model());
            if (!sortable)
                return;

            if (active_sort_model_column_ &&
                *active_sort_model_column_ == model_column) {
                if (active_phase_ == SortPhase::Ascending)
                    active_phase_ = SortPhase::Descending;
                else if (active_phase_ == SortPhase::Descending) {
                    active_sort_model_column_.reset();
                    active_phase_.reset();
                    sortable->set_sort_column(
                        Gtk::TreeSortable::DEFAULT_UNSORTED_COLUMN_ID,
                        Gtk::SORT_ASCENDING);
                    refresh_sort_indicators();
                    return;
                }
            } else {
                active_sort_model_column_ = model_column;
                active_phase_ = SortPhase::Ascending;
            }

            apply_sort();
            refresh_sort_indicators();
        }

        void apply_sort() {
            auto sortable = resolve_sortable(get_model());
            if (!sortable || !active_sort_model_column_ || !active_phase_)
                return;
            const Gtk::SortType order = *active_phase_ == SortPhase::Ascending
                                            ? Gtk::SORT_ASCENDING
                                            : Gtk::SORT_DESCENDING;
            sortable->set_sort_column(*active_sort_model_column_, order);
        }

        void refresh_sort_indicators() {
            const int n = get_n_columns();
            for (const auto &reg : sort_columns_) {
                if (reg.tree_column_index < 0 || reg.tree_column_index >= n)
                    continue;
                Gtk::TreeViewColumn *col = get_column(reg.tree_column_index);
                if (!col)
                    continue;
                col->set_sort_indicator(false);
            }
            if (!active_sort_model_column_ || !active_phase_)
                return;
            auto it = std::find_if(
                sort_columns_.begin(), sort_columns_.end(),
                [this](const SortColumnReg &r) {
                    return r.model_column == *active_sort_model_column_;
                });
            if (it == sort_columns_.end() || it->tree_column_index < 0 ||
                it->tree_column_index >= n)
                return;
            Gtk::TreeViewColumn *active = get_column(it->tree_column_index);
            if (!active)
                return;
            active->set_sort_indicator(true);
            active->set_sort_order(*active_phase_ == SortPhase::Ascending
                                      ? Gtk::SORT_ASCENDING
                                      : Gtk::SORT_DESCENDING);
        }

    private:
        enum class SortPhase { Ascending, Descending };

        struct SortColumnReg {
                int tree_column_index = -1;
                int model_column = -1;
                sigc::connection clicked_conn;
        };

        std::vector<SortColumnReg> sort_columns_;
        std::optional<int> active_sort_model_column_;
        std::optional<SortPhase> active_phase_;
        sigc::connection model_property_conn_;
};
